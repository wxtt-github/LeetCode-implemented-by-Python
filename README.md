```python
class Solution:

        left = 0
        right = len(numbers) - 1
```

```python
class Solution:
```

```python
        nums.sort()
        cnt = 0
        left = 0
        right = len(nums) - 1
        while left < right:
            s = nums[left] + nums[right]
            if s < target:
                cnt += right - left
                left += 1
            elif s >= target:
                right -= 1
        return cnt
```

4. [16. æœ€æ¥è¿‘çš„ä¸‰æ•°ä¹‹å’Œ](https://leetcode.cn/problems/3sum-closest/)

```python
"""

åœ¨åŒæŒ‡é’ˆå†…ï¼Œæ¯æ¬¡å°è¯•æ›´æ–°ç»å¯¹å€¼ï¼Œå¹¶å°½é‡å¾€ç€targetå€¼é å³å¯ã€‚éœ€è¦æ³¨æ„çš„æ˜¯
ä¼˜åŒ–ï¼Œä¸€æ–¹é¢åŒæŒ‡é’ˆéå†æ—¶è‹¥ç›¸ç­‰ï¼Œç›´æ¥è¿”å›targetï¼Œå¦ä¸€æ–¹é¢ï¼Œä¾ç„¶å¯ä»¥æ ¹æ®
ä¸targetå€¼è¿›è¡Œå¤§å°æ¯”è¾ƒè¿›è¡Œä¸¤ä¸ªä¼˜åŒ–ï¼Œä¸€ä¸ªbreakä¸€ä¸ªcontinueï¼Œå¯¹äºcontinue

æ—¶é—´å¤æ‚åº¦ï¼šO(n^2)ï¼Œå›ºå®šç¬¬ä¸€ä¸ªæ•°å’ŒåŒæŒ‡é’ˆå…±ä¸¤å±‚åµŒå¥—å¾ªç¯ã€‚

class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:
        nums.sort()
        temp = abs(nums[0] + nums[1] + nums[2] - target)
        ans = nums[0] + nums[1] + nums[2]
        length = len(nums)
        for i in range(length-2):
            x = nums[i]
            if i > 0 and nums[i] == nums[i-1]:
                continue

            # ä¼˜åŒ–1
            s = x + nums[i+1] + nums[i+2]
            if s > target:
                if s - target < temp:
                    ans = s
                break
            # ä¼˜åŒ–2
            s = x + nums[length-2] + nums[length-1]
            if s < target:
                if target - s < temp:
                    ans = s
                    temp = target - s
                continue

            left = i + 1
            right = length - 1
            while left < right:
                s1 = nums[left] + nums[right] + x
                s2 = abs(nums[left] + nums[right] + x - target)
                if s2 < temp:
                    temp = s2
                    ans = nums[left] + nums[right] + x
                if s1 < target:
                    left += 1
                elif s1 > target:
                    right -= 1
                else:
                    return target
        return ans
```

```python
class Solution:
    def triangleNumber(self, nums: List[int]) -> int:
        nums.sort(reverse=True)
        ans = 0
        length = len(nums)
        if length < 3:
            return 0
        for i in range(length-2):
            x = nums[i]
            # ä¼˜åŒ–1
            if nums[length-2] + nums[length-1] > x:
                ans += (length-i) * (length-i-1) * (length-i-2) // 6
                break
            # ä¼˜åŒ–2
            if nums[i+1] + nums[i+2] <= x:
                continue
            left = i + 1
            right = length - 1
            while left < right:
                s = nums[left] + nums[right]
                if s > x:
                    ans += right - left
                    left += 1
                elif s <= x:
                    right -= 1
        return ans
```
